\newcommand{\checkNumberModulo}[2]{%
  \ifthenelse{\modulo{#1}{#2}=0}%

  }

\makeatletter
\newcommand{\modulo}[2]{%
  \number\numexpr#1-#2*(\number\numexpr(#1/#2)\relax)\relax
}
